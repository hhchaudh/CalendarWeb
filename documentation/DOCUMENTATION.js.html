function changeMonth(i) {
    var newMonth = i;
    //var myString = i;

        newMonth = 12;
    }
    if(newMonth > 12) {
        newMonth = 1;

    //alert(myString)

    localStorage.setItem("month", newMonth);
    window.location.href = "monthlyView.html";
}

/**
 * function intToMonth(j)
 * @summary converts the given integer to the appropriate name for that month's int
 * @param j is the month's integer representation
 * @return the name of the month
 */
function intToMonth(j)
{
    var x = ""
    if (j == 1 || j == 13) {x = "January"}
    else if(j == 2) {x = "February"}
    else if(j == 3) {x = "March"}
    else if(j == 4) {x = "April"}
    else if(j == 5) {x = "May"}
    else if(j == 6) {x = "June"}
    else if(j == 7) {x = "July"}
    else if(j == 8) {x = "August"}
    else if(j == 9) {x = "September"}
    else if(j == 10) {x = "October"}
    else if(j == 11) {x = "November"}
    else if(j == 12 || j == 0) {
        x = "December"
    }
    return x
}

/**
 * @summary method to generate month specific table
 * @param (f,l) 'f' is the day of the week, i.e. Sunday, Monday, that the month begins on.
 *'l' is the number of days in that month. A table is then dynamically generated by starting to append cells
 *from the 'f' on row 0, to 'l' how many rows are needed to complete the month.
 *@return: a table is generated specific to each month integer
 */
function addTable(f, l) {
    var month = localStorage["month"]
    var monthName = intToMonth(month)


    var name = document.createElement("H")
    var text = document.createTextNode(monthName)
    name.appendChild(text);
    name.setAttribute("class","area");

    var myTableDiv = document.getElementById("metric_results")
    var table = document.createElement('TABLE')

    if(document.getElementsByClassName("monthTable").length > 0) {
        removeElementsByClass("monthTable")
    }
    table.setAttribute("class", "monthTable")
    var tableBody = document.createElement('TBODY')

    table.border = '0'
    table.appendChild(tableBody);

    var heading = new Array();
    heading[0] = "Sun"
    heading[1] = "Mon"
    heading[2] = "Tues"
    heading[3] = "Wed"
    heading[4] = "Thurs"
    heading[5] = "Fri"
    heading[6] = "Sat"


    var x = 1

    var cal = new Array()

    for (var i = 0; i < 6; i++) {
        cal[i] = new Array(7)
    }


    for (var i = 0; i < 6; i++) {
        for (var j = 0; j < 7; j++) {
            cal[i][j] = ""
        }
    }

    startOfWeek = f
    for (var i = 0; i < 6; i++) {
        for (var j = startOfWeek; j < 7; j++) {
            cal[i][j] = x
            if (7 * i + j > l + f - 2) {
                i = 6
                j = 7
            }
            x++
            startOfWeek = 0
        }
    }

    //TABLE COLUMNS
    var tr = document.createElement('TR');
    tableBody.appendChild(tr);
    for (i = 0; i < heading.length; i++) {
        var th = document.createElement('TH')
        th.width = '75';
        th.appendChild(document.createTextNode(heading[i]));
        tr.appendChild(th);
    }

    //TABLE ROWS
    for (i = 0; i < cal.length; i++) {
        var tr = document.createElement('TR');
        if (f + l >= 35 || i < 5)
            tr.setAttribute("class", "days")
        tr.setAttribute("id", "week" + i)
        tr.setAttribute("onclick", "openWeek(" + i + ")")

        for (j = 0; j < cal[i].length; j++)
        {
            var td = document.createElement('TD')
            if(cal[i][j] != "")
            {
                //td.setAttribute("class", "days")
                var dayNum = cal[i][j]
                var date = localStorage["month"] + "/" + dayNum
                for (var a = 1; a < 25; a++)
                {
                    for (var b = 0; b < 60; b++)
                    {
                        var eventVar = date + "/" + pad2(a) + ":" + pad2(b);
                        if (localStorage[eventVar] != undefined)
                        {
                            td.setAttribute("class", "event")
                        }
                    }
                }
            }
//                td.setAttribute("ID", "testTD" + j "_" + i)
            td.appendChild(document.createTextNode(cal[i][j]));
            tr.appendChild(td)
        }
        tableBody.appendChild(tr);
    }

    myTableDiv.appendChild(name);
    var s = document.createElement('br');
    var t = document.createElement('br');
    var u = document.createElement('br');
    var v = document.createElement('br');

    myTableDiv.appendChild(s);
    myTableDiv.appendChild(t);
    myTableDiv.appendChild(u);
    myTableDiv.appendChild(v);

    myTableDiv.appendChild(table);

    var w = document.createElement('br');
    var x = document.createElement('br');
    var y = document.createElement('br');
    var z = document.createElement('br');

    var previous = document.createElement('BUTTON');
    var foo = document.createTextNode('\u00A0\u00A0\u00A0\u00A0\u00A0\u00A0\u00A0\u00A0\u00A0\u00A0\u00A0\u00A0\u00A0\u00A0\u00A0\u00A0\u00A0\u00A0');
    var next = document.createElement('BUTTON');
    previous.setAttribute("class","button");
    previous.setAttribute('onclick', "changeMonth(" + (month - 1) + ")")
    next.setAttribute("class", "button");
    next.setAttribute('onclick', "changeMonth(" + (month - -1) + ")")
    //previous.setAttribute('onclick', "changeTest()")
    //"setCurrentMonth(" + month + ");window.location.href='monthlyView.html'")
    //previous.setAttribute('onclick', 'history.go(0)')
    var text = document.createTextNode("Previous Month");
    previous.appendChild(text);
    text = document.createTextNode("Next Month");
    next.appendChild(text);
    myTableDiv.appendChild(w);
    myTableDiv.appendChild(x);
    myTableDiv.appendChild(y);
    myTableDiv.appendChild(z);
    myTableDiv.appendChild(previous);
    myTableDiv.appendChild(foo);
    myTableDiv.appendChild(next);

}

                                                {
                                                    var recNum = prompt("Please enter the record number you would like to delete");
                                                    if(recNum != null && parseInt(recNum) == recNum)
                                                {
                                                    var eventVar = sessionStorage.getItem(recNum);
                                                    var x = localStorage[eventVar];
                                                    if(x != undefined)
                                                {
                                                    localStorage.removeItem(eventVar);
                                                    alert("Event deleted!")
                                                    window.location.reload();
                                                }
                                                    else
                                                {
                                                    alert("No event found at that record number!")
                                                }
                                                }
                                                    else
                                                {
                                                    alert("Invalid record number was input, so nothing was deleted.")
